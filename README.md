## Deployment

### Podman Compose

```yaml
services:
  mariadb:
    image: ghcr.io/daemonless/mariadb:latest
    container_name: mariadb
    environment:
      - PUID=1000
      - PGID=1000
      - TZ=Etc/UTC
      - MYSQL_ROOT_PASSWORD=changeme
      - MYSQL_DATABASE=mydb
      - MYSQL_USER=myuser
      - MYSQL_PASSWORD=mypassword
    volumes:
      - "/path/to/containers/mariadb:/config"
    ports:
      - 3306:3306
    restart: unless-stopped
```
